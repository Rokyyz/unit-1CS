def DNA_translator(in_protein:str)->str:
    out_protein = ''
    if in_protein =='A':
        out_protein = "T"
    if in_protein  == 'G':
        out_protein = 'C'
    if in_protein == 'T':
        out_protein = 'A'
    if in_protein == 'C':
        out_protein = 'G'
    if in_protein == 'AGCT':
        out_protein = 'TCGA'
    return out_protein

r = DNA_translator('AGCT')
print(r)
